* ##### 源码

```java
package ch05;

public class LinkList {
    private Link first;
    /**
     * 插入节点
     * @param value
     */
    public void insert(long value) {
        Link lnk = new Link(value);
        if(first == null) {
            first = lnk;
        }else {
            lnk.setNext(first);
            first = lnk;
        }
    }
    /**
     * 显示所有链接点的数据
     */
    public void displayAll() {
        Link current = first;
        while(current != null) {
            System.out.println(current.getData());
            current = current.getNext();
        }
    }
    /**
     * 查找节点
     * @param key
     * @return
     */
    public Link find(long key) {
        Link current = first;
        while(current.getData() != key) {
            if(current.getNext() == null) {
                return null;
            }
            current = current.getNext();
        }
        return current;
    }
    /**
     * 插入节点到指定位置
     * @param value
     * @param pos
     */
    public void insert(long value,int pos) {
        if(pos == 0) {
            insert(value);
        }else {
            Link current = first;
            for(int i=0;i<pos-1;i++) {
                current = current.getNext();
            }
            Link lnk = new Link(value);
            lnk.setNext(current.getNext());
            current.setNext(lnk);
        }
    }
}
```

package ch05;

public class TestLinkList {

	public static void main(String[] args) {
		// 创建一个链表
		LinkList ll = new LinkList();
		// 向链表中插入数据
		ll.insert(20);
		ll.insert(10);
		ll.insert(-10);
		ll.insert(-20);
		// 显示链表中的所有数据
		ll.displayAll();
		// 查找指定节点
		System.out.println("找到节点,数据为: "+ll.find(-20).getData());
		ll.insert(0, 0);
		System.out.println("----------------------------------------");
		ll.displayAll();
	}

}
