```java
/*
	值交换 :
	  需求1 - 定义一个函数交换两个基本类型变量的值
*/
class Demo04{
	public static void main(String[] args){
		int a = 3;
		int b = 5;
		changeValue(a,b);
		System.out.println("a = " + a + ", b = " + b); // a=3,b=5
	}
	public static void changeValue(int a,int b){
		int temp = a;
		a = b;
		b = temp;
		System.out.println("a = " + a + ", b = " + b); // a=5,b=3
	}
}
```
