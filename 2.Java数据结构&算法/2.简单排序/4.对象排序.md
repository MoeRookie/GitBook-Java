* ##### 自定义类

```java
package ch02;

public class Student {
    // 学号
    private int stuNo;
    // 姓名
    private String name;
    // 年龄
    private int age;
    // 性别
    private String gender;
    public Student(int stuNo, String name, int age, String gender) {
        this.stuNo = stuNo;
        this.name = name;
        this.age = age;
        this.gender = gender;
    }
    public int getStuNo() {
        return stuNo;
    }
    public void setStuNo(int stuNo) {
        this.stuNo = stuNo;
    }
    public String getName() {
        return name;
    }
    public void setName(String name) {
        this.name = name;
    }
    public int getAge() {
        return age;
    }
    public void setAge(int age) {
        this.age = age;
    }
    public String getGender() {
        return gender;
    }
    public void setGender(String gender) {
        this.gender = gender;
    }
    public void display() {
        System.out.println("学号:"+this.stuNo+",姓名:"+this.name+",年龄:"+this.age+",性别:"+this.gender);
    }
}
```

* ##### 源码

```java
package ch02;

public class StudentArray {
    // 数组
    private Student[] arr;
    // 数组中有效数据的大小
    private int elems;

    // 默认的构造方法
    public StudentArray() {
        arr = new Student[50];
    }
    public StudentArray(int max) {
        arr = new Student[max];
    }
    // 插入数据
    public void insert(Student student) {
        arr[elems] = student;
        elems++;
    }
    // 显示数据
    public void display() {
        for (int i = 0; i < elems; i++) {
            arr[i].display();
        }
    }
    // 按学号进行排序
    public void sortByNo() {
        int min = 0; // 默认指定下标为0的元素最小
        Student temp = null;
        for(int i = 0;i < elems-1;i++) {
            min = i; // 默认指定下标为i的元素最小
            for(int j = i+1;j < elems;j++) {
                if(arr[j].getStuNo() < arr[min].getStuNo()) {
                    min = j;
                }
            }
            // 交换
            temp = arr[i];
            arr[i] = arr[min];
            arr[min] = temp;
        }
    }
    // 按姓名排序
    public void sortByName() {
        int min = 0; // 默认指定下标为0的元素最小
        Student temp = null;
        for(int i = 0;i < elems-1;i++) {
            min = i; // 默认指定下标为i的元素最小
            for(int j = i+1;j < elems;j++) {
                if(arr[j].getName().compareTo(arr[min].getName()) < 0) {
                    min = j;
                }
            }
            // 交换
            temp = arr[i];
            arr[i] = arr[min];
            arr[min] = temp;
        }
    }
}
```
